-- Ik wil alle uitgevers die uitsluitend spellen in het Nederlands gepubliceerd hebben, op een lijstje, zodat dit mijn koopgedrag kan be√Ønvloeden.

SELECT pubid, pname FROM Publisher
WHERE pubid NOT IN (
    SELECT pubid
    FROM Publishes
    NATURAL JOIN Version
    NATURAL JOIN InLanguage
    WHERE langcode <> 'dut'
);
