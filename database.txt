CREATE TABLE Game (
  gid  	     	  integer	  not null,
  gname		  varchar(100)	  not null,	
  year		  integer	  not null,
  minplayers	  integer	  not null,
  maxplayers	  integer	  not null,
  minage	  integer	  not null); 

CREATE TABLE PatentHolder (
  phid 	     	  integer	  not null,
  phname	  varchar(100)	  not null);

CREATE TABLE Organization (
  oid  	     	  integer	  not null,
  oname		  integer	  not null);

CREATE TABLE Human (
  hid  	     	  integer	  not null,
  hname		  varchar(100)	  not null);

CREATE TABLE Publisher (
  pid  	     	  integer	  not null,
  pname		  varchar(100)	  not null);

CREATE TABLE Version (
  vid  	     	  integer	  not null,
  vname		  varchar(100)	  not null);

CREATE TABLE Language (
  langname   	  varchar(2)	  not null);

CREATE TABLE OnlineSeller (
  sid  	          integer	  not null,
  sname		  varchar(100)	  not null,
  website	  varchar(100)	  not null);

CREATE TABLE OfflineSeller (
  sid  	          integer	  not null,
  sname		  varchar(100)	  not null,
  address	  varchar(100)	  not null);

CREATE TABLE Patent (
  gid  	     	  integer	  not null,
  phid		  integer	  not null,
  patentcountry   varchar(100)	  not null);

CREATE TABLE Sport (
  oid  	     	  integer	  not null,
  gid		  integer	  not null);

CREATE TABLE Job  (
  gid  	     	  integer	  not null,
  hid		  integer	  not null,
  job		  varchar(10)	  not null);

CREATE TABLE Lang (
  langname   	  integer	  not null,
  vid		  integer	  not null);

CREATE TABLE Sell (
  sid  	     	  integer	  not null,
  vid		  integer	  not null,
  price		  integer	  not null);
